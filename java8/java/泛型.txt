public class Pair<T>{ //T放在类名的后边

}

泛型方法
class ArrayAlg{

 public static<T> T getMiddle(T...a){//修饰符的后边，返回值的前边
   
 }
}
调用泛型方法
ArrayAlg.<String>.getMiddle("A","B","C");

  public static<T extends Comparable> T min(T[] a){//这样T的变量就可以使用compareTo的方法
      
  }
}
